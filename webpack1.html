    mock/index.js

    const fs = require("fs");
const path = require("path");

module.exports = (app) => {
  app.all(/.*.do$/, (req, res) => {
    const JSFilePath = path.join(
      __dirname,
      `/api/v1${req.path.replace("/api/", "/").replace(".do", ".js")}`
    );
    if (fs.existsSync(JSFilePath)) {
      delete require.cache[JSFilePath];
      require(JSFilePath)(req, res);
    } else {
      res.send({
        code: "0",
        data: {},
        msg: "请求成功",
      });
    }
  });
};
